unciv_can_mobilize_yay = {
	trigger = {
		NOT = {
			civilized = yes
		}
	}
	mobilisation_size = 0.08 #the base debuff so uncivs cant mob is -0.10, so this means that if u are unciv you can mob, just less than a civilized nation
}
